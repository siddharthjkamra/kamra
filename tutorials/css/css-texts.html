Text Alignment


				

/* Text Align */

h1 {
  text-align: center;
}

h2 {
  text-align: left;
}

h3 {
  text-align: right;
}

h3 {
  text-align: justify;
}



				

/* Last Line Align */

p.x {
  text-align-last: right;
}

p.y {
  text-align-last: center;
}

p.z {
  text-align-last: justify;
}



				

/* Text Direction
Unicode-bidi changes direction */

p {
  direction: rtl;
  unicode-bidi: bidi-override;
}



				

/* Vertical alignment */

img.a {
  vertical-align: baseline;
}

img.b {
  vertical-align: text-top;
}

img.c {
  vertical-align: text-bottom;
}

img.d {
  vertical-align: sub;
}

img.e {
  vertical-align: super;
}

Text Spacing


				

/* Text Indentation */

  text-indent: 50px;
}



				

/* Letter Spacing */

h1 {
  letter-spacing: 5px;
}

h2 {
  letter-spacing: -2px;
}



				

/* Line Height */

p.small {
  line-height: 0.8;
}

p.big {
  line-height: 1.8;
}



				

/* Word Spacing */				
				
p.one {
  word-spacing: 10px;
}

p.two {
  word-spacing: -2px;
}				



				

/* White Space */

p {
  white-space: nowrap;
}
